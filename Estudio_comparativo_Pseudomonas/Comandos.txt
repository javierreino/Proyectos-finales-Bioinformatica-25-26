# CÓDIGO PYTHON - MATRIZ DE PRESENCIA/AUSENCIA DE GENES
import pandas as pd
import json
import glob

# Definir cepas y genes de interés
cepas = ['PA14', 'PAO1', 'SBW25']
genes_interes = ['NRPS_core', 'transportador_ABC', 'regulador', 'sintetasa_accesoria', 
                 'receptor', 'permeasa', 'tioseterasa', 'oxidoreductasa']

# Crear matriz vacía
M = pd.DataFrame(0, index=cepas, columns=genes_interes)

# Procesar resultados de antiSMASH para cada cepa
for cepa in cepas:
    archivos_json = glob.glob(f'antismash_{cepa}/*.json')
    for archivo in archivos_json:
        with open(archivo, 'r') as f:
            data = json.load(f)
            # Extraer regiones de sideróforos
            for region in data.get('records', []):
                if 'NRP-metallophore' in str(region.get('products', [])):
                    # Marcar presencia de genes centrales
                    M.loc[cepa, 'NRPS_core'] = 1
                    
                    # Buscar genes accesorios
                    for gene in region.get('genes', []):
                        if 'ABC' in gene.get('function', ''):
                            M.loc[cepa, 'transportador_ABC'] = 1
                        if 'regulator' in gene.get('function', ''):
                            M.loc[cepa, 'regulador'] = 1
                        # ... etc

# Guardar matriz
M.to_csv('matriz_presencia_ausencia_sideroforos.csv')
print("Matriz generada:")
print(M)


# CÓDIGO PYTHON - HEATMAP DE PRESENCIA/AUSENCIA
import seaborn as sns
import matplotlib.pyplot as plt
import pandas as pd

# Cargar matriz
M = pd.read_csv('matriz_presencia_ausencia_sideroforos.csv', index_col=0)

# Crear heatmap
plt.figure(figsize=(10, 6))
ax = sns.heatmap(M, annot=True, cmap="YlGnBu", linewidths=.5, 
                 cbar_kws={'label': 'Presencia (1) / Ausencia (0)'})

plt.title('Presencia de Genes del Operón de Sideróforos en Pseudomonas spp.', fontsize=14)
plt.xlabel('Genes', fontsize=12)
plt.ylabel('Cepas', fontsize=12)
plt.tight_layout()
plt.savefig('heatmap_sideroforos.png', dpi=300)
plt.show()
