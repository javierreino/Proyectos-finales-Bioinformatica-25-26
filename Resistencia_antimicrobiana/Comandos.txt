# CÓDIGO PYTHON - CÁLCULO DE MATRICES P-DISTANCE
from Bio import AlignIO
from Bio.Phylo.TreeConstruction import DistanceCalculator
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

# Función para calcular matriz p-distance
def calcular_matriz_pdistance(archivo_alineamiento, nombre_gen):
    # Cargar alineamiento
    alineamiento = AlignIO.read(archivo_alineamiento, "fasta")
    
    # Crear calculador de distancias (modelo 'identity' = p-distance)
    calculator = DistanceCalculator('identity')
    distance_matrix = calculator.get_distance(alineamiento)
    
    # Convertir a DataFrame
    nombres = [record.id[:15] for record in alineamiento]  # IDs truncados
    df = pd.DataFrame(distance_matrix.matrix, index=nombres, columns=nombres)
    
    # Guardar matriz
    df.to_csv(f"matrices/{nombre_gen}_pdistance.csv")
    
    # Crear heatmap
    plt.figure(figsize=(12, 10))
    sns.heatmap(df, annot=True, fmt='.3f', cmap='YlOrRd', 
                xticklabels=True, yticklabels=True)
    plt.title(f'Matriz p-distance - Gen {nombre_gen}')
    plt.tight_layout()
    plt.savefig(f"figuras/heatmap_{nombre_gen}.png", dpi=300)
    plt.close()
    
    return df

# Calcular para cada gen
genes = ['blaKPC', 'blaIMP', 'blaTEM', 'tetc']
for gen in genes:
    print(f"Calculando p-distance para {gen}...")
    df = calcular_matriz_pdistance(f'alineamientos/{gen}_alineado.fasta', gen)
    print(f"Valores mín: {df.min().min():.3f}, máx: {df.max().max():.3f}")
    print()

# CÓDIGO PYTHON - HEATMAPS PERSONALIZADOS PARA CADA GEN
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt
import numpy as np

# Configuración global
plt.rcParams['figure.figsize'] = (14, 12)
plt.rcParams['font.size'] = 10

# Función para crear heatmap mejorado
def crear_heatmap_mejorado(archivo_matriz, titulo, archivo_salida):
    # Cargar matriz
    df = pd.read_csv(archivo_matriz, index_col=0)
    
    # Crear máscara para triángulo superior (opcional)
    mask = np.triu(np.ones_like(df, dtype=bool))
    
    # Configurar heatmap
    plt.figure(figsize=(16, 14))
    
    # Heatmap con anotaciones
    sns.heatmap(df, 
                annot=True, 
                fmt='.3f',
                cmap='YlOrRd',  # Amarillo-Naranja-Rojo (mejor contraste)
                linewidths=0.5,
                linecolor='gray',
                cbar_kws={'label': 'p-distance', 'shrink': 0.8},
                mask=mask,  # Solo mostrar triángulo inferior
                square=True)
    
    plt.title(titulo, fontsize=16, pad=20)
    plt.xlabel('Secuencias', fontsize=12)
    plt.ylabel('Secuencias', fontsize=12)
    plt.xticks(rotation=45, ha='right')
    plt.yticks(rotation=0)
    
    plt.tight_layout()
    plt.savefig(archivo_salida, dpi=300, bbox_inches='tight')
    plt.show()
    plt.close()

# Crear heatmaps para cada gen
crear_heatmap_mejorado('matrices/blaIMP_pdistance.csv', 
                      'Matriz p-distance - Gen blaIMP (mayor variabilidad: hasta 60%)',
                      'figuras/heatmap_blaIMP_final.png')

# CÓDIGO PYTHON - ANÁLISIS COMPARATIVO DE VARIABILIDAD
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import numpy as np

# Cargar todas las matrices
genes = ['blaKPC', 'blaIMP', 'blaTEM', 'tetc']
estadisticas = []

for gen in genes:
    df = pd.read_csv(f'matrices/{gen}_pdistance.csv', index_col=0)
    
    # Obtener valores de la matriz (sin diagonales)
    valores = df.values
    np.fill_diagonal(valores, np.nan)
    valores = valores[~np.isnan(valores)]
    
    estadisticas.append({
        'Gen': gen,
        'Media': np.mean(valores),
        'Mediana': np.median(valores),
        'Mínimo': np.min(valores),
        'Máximo': np.max(valores),
        'Desv_Est': np.std(valores),
        'Rango': np.max(valores) - np.min(valores)
    })
    
    print(f"\n=== Gen {gen} ===")
    print(f"Media: {np.mean(valores):.4f}")
    print(f"Máximo: {np.max(valores):.4f}")
    print(f"Interpretación: {np.max(valores)*100:.1f}% divergencia máxima")

# Crear DataFrame de estadísticas
df_est = pd.DataFrame(estadisticas)
df_est.to_csv('estadisticas_variabilidad.csv', index=False)
print("\n=== RESUMEN DE VARIABILIDAD ===")
print(df_est.to_string(index=False))

# Gráfico comparativo
plt.figure(figsize=(10, 6))
sns.barplot(data=df_est, x='Gen', y='Máximo', palette='viridis')
plt.title('Máxima divergencia por gen (p-distance)')
plt.ylabel('p-distance máxima')
plt.xlabel('Gen')
plt.axhline(y=0.6, color='r', linestyle='--', label='60% (blaIMP)')
plt.axhline(y=0.013, color='g', linestyle='--', label='1.3% (tetC)')
plt.legend()
plt.savefig('figuras/comparacion_variabilidad.png', dpi=300)
plt.show()
