# Crear entorno conda para el proyecto
conda create -n resistome_pipeline python=3.9
conda activate resistome_pipeline

# Instalar herramientas bioinformáticas
conda install -c bioconda entrez-direct
conda install -c bioconda mafft
conda install -c bioconda clustalo
conda install -c bioconda iqtree
conda install -c bioconda fasttree
conda install -c bioconda emboss
conda install -c bioconda ncbi-acc-download
conda install -c bioconda seqkit

# Instalar librerías de Python para análisis y visualización
conda install -c conda-forge biopython
conda install -c conda-forge pandas numpy matplotlib seaborn jupyter
conda install -c conda-forge scipy scikit-learn
