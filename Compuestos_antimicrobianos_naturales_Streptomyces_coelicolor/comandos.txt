# CÓDIGO PYTHON - EXTRACCIÓN DE GENES DE INTERÉS
from Bio import SeqIO
import pandas as pd

# Cargar genoma
genoma = SeqIO.read("Streptomyces_coelicolor_A3_2.gb", "genbank")

# Definir regiones de interés (basado en antiSMASH)
regiones_interes = [
    {"nombre": "actinorhodin", "start": 5494930, "end": 5567424, "tipo": "T2PKS"},
    {"nombre": "undecylprodigiosin", "start": 6423650, "end": 6476442, "tipo": "prodigiosin"},
    {"nombre": "coelichelin", "start": 493989, "end": 552325, "tipo": "NRP-metallophore"},
    {"nombre": "CDA", "start": 3523335, "end": 3603988, "tipo": "NRPS"}
]

# Extraer cada región
for region in regiones_interes:
    seq_region = genoma[region["start"]:region["end"]]
    seq_region.id = f"S_coelicolor_{region['nombre']}"
    seq_region.description = f"{region['tipo']} cluster {region['start']}-{region['end']}"
    SeqIO.write(seq_region, f"cluster_{region['nombre']}.fasta", "fasta")
    print(f"ExtraÃ­do: cluster_{region['nombre']}.fasta - {len(seq_region)} pb")

# Crear tabla de resumen
df = pd.DataFrame(regiones_interes)
df.to_csv("clusters_s_coelicolor.csv", index=False)
print("\nTabla de clusters guardada en clusters_s_coelicolor.csv")
